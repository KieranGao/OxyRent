Dockerfile示例

# C++服务Dockerfile示例
FROM ubuntu:22.04 AS builder

# 安装依赖
RUN apt-get update && apt-get install -y \
    g++ cmake make \
    libboost-all-dev libprotobuf-dev \
    libgrpc++-dev protobuf-compiler-grpc \
    libhiredis-dev libmysqlcppconn-dev

# 复制源码
WORKDIR /app
COPY . .

# 编译
RUN mkdir build && cd build \
    && cmake .. && make -j$(nproc)

# 运行阶段
FROM ubuntu:22.04
COPY --from=builder /app/build/server /usr/local/bin/
COPY config.ini /etc/server/config.ini

EXPOSE 8080
CMD ["server", "--config", "/etc/server/config.ini"]

启动命令

# 构建并启动所有服务
docker-compose up -d --build

# 查看服务状态
docker-compose ps

# 查看日志
docker-compose logs -f [service_name]

# 停止所有服务
docker-compose down

# 清理数据卷（谨慎使用）
docker-compose down -v

优势

  • 环境一致性：开发、测试、部署环境完全一致
  • 快速部署：一键启动所有服务
  • 易于扩展：可以轻松添加新服务或扩展现有服务
  • 资源隔离：每个服务独立运行，互不影响
  • 便于演示：课程设计答辩时可以快速展示系统
